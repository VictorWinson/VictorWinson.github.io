UNIX basic commands

基础命令#

ls

mkdir 

rm

rmdir

cd 

touch

cat

echo

grep

|

>

>>

ln (-s) 

find <起始目录> <选项表达式> <条件匹配表达式> <动作表达式>

# 选项表达式
-L 遇到符号链接，跟踪链接地址
-regextype (-regex/-iregex)
-depth/-d
-moount/-xdev 查找时不跨越文件系统
-maxdepth LEVEL 最大查找深度

# 条件匹配表达式
-name
myfile
'*myfile*'

-iname 文件名含有
-amin access N minutes ago
-atime access N days ago
-cmin create N minutes ago
-ctime create N days ago
-mmin modify N minutes ago
-mtime modify N days ago
#    访问文件：vim/sh/bash/cat/more等操作会重置Access time
#    读取文件属性：ls/stat/wc/du等操作不会重置Access time
#    修改文件：会重置Modify time 和 Change time
#    修改权限但不调整内容： 不会重置Modify time， 但会重置 Change time。
-used used in N minutes
-newer
-uid
-gid
-user 
-group
-nouser 所属不在/etc/passwd
-nogroup 所属组不在/etc/group
-perm MODE 权限为MODE
-perm -MODE 权限至少为MODE
-perm /MODE或+MODE 有一个用户权限为MODE即可
-type b c d p f l s
-links N 硬链接数量
-inum N inode为N
-samefile NAME 查找与NAME文件具有相同inode的文件
-size
-empty

#动作表达式
-print
-ls
-prune
-exec COMMAND {} \;
-ok COMMAND {} \;

# 用户管理
adduser

# 网络

搜索#

find#

  • 实时扫描

fg#

  • 比find更快，用数据库，类似于windows下的everything

结果整理#

rg#

  • 搜索工具,比grep快,可以搜索流或者文件内容

sed#

  • 按正则匹配筛选

awk#

  • 按字符块输出, 支持正则筛选

cut#

比较#

diff#

  • 比较文件或目录

    cmp#

  • 比较两个文件并且定位每个不同的位置

    sdiff#

  • 比较文件并且可以合并

    wdiff#

  • 一个单词一个单词比较不同

    mntui#

  • 命令行ui，适合在没有DE的情况下配置wifi，依赖NetworkManager。

优化cpu调度#

tlp#

使用tlpui
AC/BAT调度模式分别设置为schedutil和powersave
关闭BAT状态下的boost

禁用不必要服务项及应用#

systemctl #查看所有服务项  
systemctl #stop 查看临时禁用是否有影响  
systemctl #disable 禁用开机启动  
systemctl #mask 阻止关联启动  

kill $(pidof {the programe you want to stop})
# 我禁用了compton(窗口渲染器)

限制进程cpu资源分配#

  • cpulimit
    cpulimit -e {nameOfProgram} -l 30
# -l 的最大值是cpu线程数
