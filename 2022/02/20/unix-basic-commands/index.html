UNIX basic commands

基础命令#

ls

mkdir 

rm

rmdir

cd 

touch

cat

echo

grep

|

>

>>

ln (-s) 

find <起始目录> <选项表达式> <条件匹配表达式> <动作表达式>

# 选项表达式
-L 遇到符号链接，跟踪链接地址
-regextype (-regex/-iregex)
-depth/-d
-moount/-xdev 查找时不跨越文件系统
-maxdepth LEVEL 最大查找深度

# 条件匹配表达式
-name
myfile
'*myfile*'

-iname 文件名含有
-amin access N minutes ago
-atime access N days ago
-cmin create N minutes ago
-ctime create N days ago
-mmin modify N minutes ago
-mtime modify N days ago
#    访问文件：vim/sh/bash/cat/more等操作会重置Access time
#    读取文件属性：ls/stat/wc/du等操作不会重置Access time
#    修改文件：会重置Modify time 和 Change time
#    修改权限但不调整内容： 不会重置Modify time， 但会重置 Change time。
-used used in N minutes
-newer
-uid
-gid
-user 
-group
-nouser 所属不在/etc/passwd
-nogroup 所属组不在/etc/group
-perm MODE 权限为MODE
-perm -MODE 权限至少为MODE
-perm /MODE或+MODE 有一个用户权限为MODE即可
-type b c d p f l s
-links N 硬链接数量
-inum N inode为N
-samefile NAME 查找与NAME文件具有相同inode的文件
-size
-empty

#动作表达式
-print
-ls
-prune
-exec COMMAND {} \;
-ok COMMAND {} \;

# 用户管理
adduser

addgroup

usermod

# 网络
ifconfig

# down load a url/*
wget -m -np -nH
